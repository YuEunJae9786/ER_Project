-- 제품 리뷰 게시판 테이블
CREATE TABLE PRODUCTREVIEW(
    RNO NUMBER PRIMARY KEY,
    WRITER VARCHAR2(30),
    TITLE VARCHAR2(30),
    CONTENT VARCHAR2(100),
    STAR NUMBER,
    REGDATE DATE DEFAULT SYSDATE,
    UPDATEDATE DATE DEFAULT SYSDATE,
    PCODE VARCHAR2(30)
    HELPCOUNT NUMBER
);

CREATE SEQUENCE PRODUCTREVIEW_SEQ INCREMENT BY 1 START WITH 1 NOCYCLE NOCACHE;

-- 제품 정보 테이블
CREATE TABLE PRODUCTINFO(
    PCODE VARCHAR2(30) PRIMARY KEY NOT NULL,
    PNAME VARCHAR2(30),
    CPNAME VARCHAR2(100),
    PBASICINFO VARCHAR2(1500),
    PCONTENT VARCHAR2(3000),
    PSPEC1 VARCHAR2(100),
    PSPEC2 VARCHAR2(100),
    PSPEC3 VARCHAR2(100),
    PIMG1 VARCHAR2(100),
    PIMG2 VARCHAR2(100),
    PIMG3 VARCHAR2(100),
    PIMG4 VARCHAR2(100),
    PIMG5 VARCHAR2(100),
    PINFOIMG1 VARCHAR2(100),
    PINFOIMG2 VARCHAR2(100)
);
