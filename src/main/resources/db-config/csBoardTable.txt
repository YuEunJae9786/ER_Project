## NOTICE 공지사항 게시판
CREATE TABLE shm.NOTICE(
    NOTICE_NO int(10) PRIMARY KEY AUTO_INCREMENT,
    NOTICE_WRITER VARCHAR(30) NOT NULL,
    NOTICE_TITLE VARCHAR(100) NOT NULL,
    NOTICE_CONTENT VARCHAR(3000),
    NOTICE_VIEW int(10),
    NOTICE_REGDATE DATETIME DEFAULT CURRENT_TIMESTAMP,
    NOTICE_UPDATEDATE DATETIME DEFAULT CURRENT_TIMESTAMP
);

## NOTICE_IMAGE 공지사항 이미지 게시판 
CREATE TABLE shm.NOTICE_IMAGE(
    NI_NO INT(10) PRIMARY KEY AUTO_INCREMENT,
    NOTICE_NO INT(10) NOT NULL,
    NI_PATH VARCHAR(100) NOT NULL,
    NI_NAME VARCHAR(100) NOT NULL
);

## FAQ 자주묻는 질문 게시판
CREATE TABLE shm.FAQ (
    FAQ_NO INT(10) PRIMARY KEY AUTO_INCREMENT,
    FAQ_WIRTER VARCHAR(30) NOT NULL,
    FAQ_TITLE VARCHAR(100) NOT NULL,
    FAQ_CONTENT VARCHAR(3000) NOT NULL,
    FAQ_VIEW INT(10) DEFAULT 0,
    FAQ_MAINCATEGORI VARCHAR(30) NOT NULL,
    FAQ_REGDATE DATETIME DEFAULT CURRENT_TIMESTAMP,
    FAQ_UPDATEDATE DATETIME DEFAULT CURRENT_TIMESTAMP
);

## FAQ_IMAGE 자주묻는 질문 이미지 게시판
CREATE TABLE shm.FAQ_IMAGE(
    FI_NO INT(10) PRIMARY KEY AUTO_INCREMENT,
    FAQ_NO INT(10) NOT NULL,
    FI_PATH VARCHAR(100) NOT NULL,
    FI_NAME VARCHAR(100) NOT NULL
);


## QNA 1:1 질문 게시판
CREATE TABLE shm.QNA(
    QNA_NO INT(10) PRIMARY KEY AUTO_INCREMENT,
    QNA_WRITER VARCHAR(30) NOT NULL,
    QNA_TITLE VARCHAR(100) NOT NULL,
    QNA_CONTENT VARCHAR(3000),
    QNA_MAINCATEGORI VARCHAR(30) NOT NULL,
    QNA_TOW VARCHAR(30) NOT NULL,
    QNA_VIEW INT(10) DEFAULT 0,
    QNA_ISANSWER CHAR(1) DEFAULT 'F' CHECK( QNA_ISANSWER IN('Y', 'F') ),
    QNA_REGDATE DATETIME DEFAULT CURRENT_TIMESTAMP,
    QNA_UPDATEDATE DATETIME DEFAULT CURRENT_TIMESTAMP
);


## QNA 답변 게시판
CREATE TABLE shm.QNA_ANSWER(
    QA_NO INT(10) PRIMARY KEY AUTO_INCREMENT,
    QNA_NO INT(10) NOT NULL,
    QA_WRITER VARCHAR(30) NOT NULL,
    QA_CONTENT VARCHAR(3000) NOT NULL,
    QA_REGDATE DATETIME DEFAULT CURRENT_TIMESTAMP
);
